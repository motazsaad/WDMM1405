class Circle:
    def __init__(self, c=(0,0), r=0):
        self.center = c 
        self.radius = r 
    def get_aria(self):
        return self.radius * self.radius * 3.14
    def get_perimeter(self):
        return 2 * self.radius * 3.14 
    def __str__(self):
        return 'center: {}\traduis:{}'.format(self.center, self.radius)
